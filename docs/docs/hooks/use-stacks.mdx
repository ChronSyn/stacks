It returns the `Stacks` context object.

<Tabs {...defaultTabsProps}>
<TabItem value="typescript">

```tsx
declare type Breakpoints = {
  readonly tablet: number
  readonly desktop: number

declare type Context = {
  readonly debug: boolean
  readonly spacing: number

  readonly dimensions: ReactNative.ScaledSize
```

</TabItem>
<TabItem value="rescript">

```res
  tablet: float,
  desktop: float,
}

type context = {
  debug: bool,
  spacing: float,

  dimensions: ReactNative.Dimensions.displayMetrics,
```

</TabItem>

## Example

```jsx live
function() {
  const Card = props => {
    const context = useStacks()

    return (
      <Text>{JSON.stringify(context)}</Text>
    )
  }

  return (
    <StacksProvider spacing={4}>
      <Card />
    </StacksProvider>
  )
}
```
