import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';
import { defaultTabsProps } from '../../core/components';

It generates a style object with a random `backgroundColor` value.

<Tabs {...defaultTabsProps}>
<TabItem value="typescript">

```tsx
declare const useDebugStyle: () => ReactNative.StyleProp<ReactNative.ViewStyle> | undefined | null
```

</TabItem>
<TabItem value="rescript">

```res
let useDebugStyle: unit => option<ReactNative.Style.t>
```

</TabItem>
</Tabs>

## Example

```jsx live
function() {
  const Card = () => {
    const debugStyle = useDebugStyle()

    return (
      <View style={debugStyle}>
        <Text>Debug</Text>
      </View>
    )
  }

  return (
    <StacksProvider debug={true}>
      <Card />
    </StacksProvider>
  )
}
```
